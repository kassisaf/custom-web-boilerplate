def playlist(songs):
    valid_pairs = 0
    for i in range(len(songs) - 1):
        for j in range(i + 1, len(songs)):
            if (songs[i] + songs[j]) % 60 == 0:
                valid_pairs += 1
                print(f'{i} + {j}: {songs[i]} + {songs[j]}')
    return valid_pairs
